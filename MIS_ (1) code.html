to move-educated-people

  ;pitamo ljude koji se trenutno ne skoluju i zavrsili su barem 3 godine edu.
  ask people with [studying? = false and studying-finished-ticks - studying-started-ticks >= 36]
  [
    set heading (heading + 45 - random 90)
    ifelse knowledge-level < min-knowledge
    [
      if [pcolor] of patch-ahead 1 != black
      [
      let nearest-black-patch min-one-of patches with [pcolor = black][ distance myself ]
      face nearest-black-patch
      ]
      forward 1
    ]
    [
      let nearest-biotech-center min-one-of biotech-centers [ distance myself ]
      let x [pxcor] of nearest-biotech-center
      let y [pycor] of nearest-biotech-center
      ;if [count people with [working = true] in-radius 2] of patch x y  > 6 [output-print [count people with [working = true] in-radius 2] of patch x y]
        ifelse distance nearest-biotech-center > 2
        [
        ifelse distance nearest-biotech-center < 14 and [count people with [working = true] in-radius 2] of patch x y  <= 5
        [

          setxy x y
          set working true

        ]
        [
            set heading (heading + 45 - random 90)
            if [pcolor] of patch-ahead 1 != black
            [
              let nearest-black-patch min-one-of patches with [pcolor = black][ distance myself ]
              face nearest-black-patch
            ]
            forward 1
        ]
        ]
      [
      if working = false
        [
                    if [pcolor] of patch-ahead 1 != black
            [
              let nearest-black-patch min-one-of patches with [pcolor = black][ distance myself ]
              face nearest-black-patch
            ]
            forward 1
        ]
      ]
        ;set label round knowledge-level ; for DEBUG purposes only - to remove in production


    ]
  ]
end

to move-uneducated-people

  ask people with [studying? = false and studying-finished-ticks - studying-started-ticks < 36]
  [
    let nearest-center min-one-of edu-centers [ distance myself ]
    ifelse distance nearest-center < 14
    [
      ifelse ((random (100 - min-willingness)) + min-willingness) > learning-willingness
      [rt random 180 + 90]
      [

      let x [pxcor] of nearest-center
      let y [pycor] of nearest-center
      setxy x y
      ]
    ]
    [set heading (heading + 45 - random 90)]


    forward 1
  ]
end

to improving-people-biotech-knowledge
  ask people with [knowledge-level > min-knowledge and biotech-knowledge-level < 100 and working = true]
    [
        if count biotech-centers in-radius 2 > 0
        [
        let nearest-center min-one-of biotech-centers [ distance myself ]
        let knowledge-to-absorb-factor (sqrt (learning-ability * learning-willingness)) / 100
        let knowledge-to-absorb knowledge-to-absorb-factor * [available-knowledge-per-year] of nearest-center



        let KTAnormalized ((b - a) * ((knowledge-to-absorb - min-value) / (max-value - min-value)) + a)
        let knowledge-growth (42 / working-years + 3)
        if knowledge-growth > 10 [set knowledge-growth 10]
        let total-knowledge-growth KTAnormalized * knowledge-growth

        set biotech-knowledge-level biotech-knowledge-level + (total-knowledge-growth / 12)

        if biotech-knowledge-level > 11 [set label round biotech-knowledge-level]

         if ticks mod 12 = 0 [set working-years working-years + 1]

      ]
    ]
end


to creating-new-biotech-center

  let people-in-biotech-centers 0
  ask biotech-centers[set people-in-biotech-centers people-in-biotech-centers + count people in-radius 2]


  if count people with [working = true] >= count biotech-centers * 6
  and count people with [biotech-knowledge-level > 70] > count biotech-centers * 2
  and count biotech-centers < count edu-centers
  [
    create-biotech-centers 1[
    set shape "box"
    set color 105
    set size 1.2
    set radius 1 + random 3
    set knowledge-level  40 + random 61
    set available-knowledge-per-year knowledge-level / 10
    set ticks-without-master 0

    let x 0
    let y 0
    while [any? other edu-centers in-radius 4 or any? other biotech-centers in-radius 4][
        set x round random-xcor
        set y round random-ycor
        setxy x y
      ]
    ask patches in-radius 2 [set pcolor sky + 4]

    ;jednog agenta sa preko 70 biotech znanja saljemo u novi biotech centar
    ask people with [biotech-knowledge-level > 70]
    [
        if custom-stop = false
        [
          set custom-stop true
          setxy x y
        ]
      ]

    ]
    set custom-stop  false

 ]
end

to destroy-biotech-center
;treba doradit uslov
;ako nemamo barem 1 agenta sa preko 70 znanja unistavamo ga(nakon 72 tika)
ask biotech-centers
[
    ;how many people work there...
    set label count people with [working = true] in-radius 2

    ifelse count people with [biotech-knowledge-level > 70] in-radius 2 < 1
    [
      set ticks-without-master  ticks-without-master + 1
      if ticks-without-master >= 72 and count biotech-centers > 1
      [
        ask patches in-radius 2 [set pcolor black]
        die
      ]
    ]
    [set ticks-without-master 0]
]
end

to check-for-transfer-biotech-knowledge
  ; vrati sve biotech inzenjere sa nivoem znanja vecim od 40
  ask people with [biotech-knowledge-level > 40]
  [
    ; i pohrani njihov nivo znanja u temp varijablu
    let biotech-knowledge-of-current-agent biotech-knowledge-level

    ; provjeri da li ima neko tu na istom patch-u osim trenutnog agenta, a da ima manje znanja od trenutnog
    if any? other people-here with [biotech-knowledge-level < biotech-knowledge-of-current-agent]
    [
      ; prodji kroz te agente sa manje znanja i povecaj im znanje koristeci temp formulu
      ask other people-here with [biotech-knowledge-level < biotech-knowledge-of-current-agent]
      [
        set biotech-knowledge-level biotech-knowledge-level + (biotech-knowledge-level / sqrt (1 + (biotech-knowledge-level ^ 2)))

      ]
    ]
  ]
end

to aging-people

  if ticks mod 12 = 0 [
    ask people [
      set age age + 1
    ]
  ]

end


to dying
  ask people [

    if age > 65 and age < 75 [
      if coin-flip? [
        set death-count death-count + 1
        die
      ]
    ]
    if age >= 75 [
      set death-count death-count + 1
      die
    ]

  ]
end

to new-generation

  if ticks mod 12 = 0 [
    creating-people ((count people + death-count) / 100 * 1 + death-count) 0
    set death-count 0
  ]
end

; destroying edu-centers and agents
to cataclysm

  let educated people with [studying-finished-ticks - studying-started-ticks >= 36]
  let uneducated people with [studying-finished-ticks - studying-started-ticks < 36]

  let educated-to-kill kill-educated-percentage / 100 * count educated
  let uneducated-to-kill kill-uneducated-percentage / 100 * count uneducated

  ask n-of educated-to-kill educated [die]
  ask n-of uneducated-to-kill uneducated [die]
  ask n-of (kill-eduCenters-percentage / 100 *  count edu-centers)  edu-centers [
    ask patches in-radius 2 [set pcolor black]
    die
  ]

  ;-----------------------------------------------------------------------------------------
  ask n-of (kill-eduCenters-percentage / 100 *  count biotech-centers)  biotech-centers [
    ask patches in-radius 2 [set pcolor black]
    die
  ]
  ;-----------------------------------------------------------------------------------------

  ask edu-centers [
    set knowledge-level knowledge-level * (1 - ((kill-eduCenters-knowledge-percentage - random (kill-eduCenters-knowledge-percentage / 2)) / 100))
  ]
    ask biotech-centers [
    set knowledge-level knowledge-level * (1 - ((kill-BioTechCenters-knowledge-percentagekill - random (kill-BioTechCenters-knowledge-percentagekill / 2)) / 100))
  ]
  ;kill-BioTechCenters-knowledge-percentagekill
  set educated-cataclysm-deaths educated-to-kill + educated-to-kill
  set uneducated-cataclysm-deaths uneducated-to-kill + uneducated-to-kill
end
